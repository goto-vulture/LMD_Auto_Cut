\newcommand{\drawCoordSystem}[5][black]{%
    \def\xmin{#2}
    \def\xmax{#3}
    \def\ymin{#4}
    \def\ymax{#5}

    \draw[->, thick, #1] (\xmin, 0) -- (\xmax, 0);
    \draw[->, thick, #1] (0, \ymin) -- (0, \ymax);

    % Ursprung markieren
    \fill[#1] (0,0) circle (0.1);

    \foreach \x in {\xmin, ..., \xmax}
    {
        \draw[dotted, #1] (\x, \ymin) -- (\x, \ymax) node[above] {\x};
    }
    \foreach \y in {\ymin, ..., \ymax}
    {
        \draw[dotted, #1] (\xmin, \y) -- (\xmax, \y) node[right] {\y};
    }

    \draw[thick, #1] (-\xmax, 0) node[left] {x};
    \draw[thick, #1] (0, -\ymax) node[below] {y};
}
